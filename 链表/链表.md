### 1. 链表基本定义

首先，定义链表节点的类：

```python
class ListNode:
    def __init__(self, value=0, next=None):
        self.value = value
        self.next = next
```

### 2. 创建链表

创建一个链表通常需要一个头节点，表示链表的起始点。头节点的`next`指向链表中的第一个元素。

```python
def create_linked_list(values):
    if not values:
        return None

    head = ListNode(values[0])
    current = head

    for value in values[1:]:
        current.next = ListNode(value)
        current = current.next

    return head
```

### 3. 打印链表

```python
def print_linked_list(head):
    current = head
    while current:
        print(current.value, end=" -> ")
        current = current.next
    print("None")
```

### 4. 插入节点

#### 4.1 在指定位置插入节点

```python
def insert_at_position(head, position, value):
    if position < 0:
        return head

    new_node = ListNode(value)

    if position == 0:
        new_node.next = head
        return new_node

    current = head
    for _ in range(position - 1):
        if current is None:
            break
        current = current.next

    if current is None:
        return head

    new_node.next = current.next
    current.next = new_node

    return head
```

#### 4.2 在末尾插入节点

```python
def insert_at_end(head, value):
    new_node = ListNode(value)

    if not head:
        return new_node

    current = head
    while current.next:
        current = current.next

    current.next = new_node
    return head
```

### 5. 删除节点

#### 5.1 删除指定位置的节点

```python
def delete_at_position(head, position):
    if position < 0 or not head:
        return head

    if position == 0:
        return head.next

    current = head
    for _ in range(position - 1):
        if current is None or current.next is None:
            break
        current = current.next

    if current is None or current.next is None:
        return head

    current.next = current.next.next
    return head
```

#### 5.2 删除指定值的节点

```python
def delete_node_with_value(head, value):
    if not head:
        return None

    if head.value == value:
        return head.next

    current = head
    while current.next and current.next.value != value:
        current = current.next

    if current.next:
        current.next = current.next.next

    return head
```

### 6. 反转链表

```python
def reverse_linked_list(head):
    prev = None
    current = head

    while current:
        next_node = current.next
        current.next = prev
        prev = current
        current = next_node

    return prev
```

### 7. 使用示例

```python
values = [1, 2, 3, 4, 5]
head = create_linked_list(values)
print("Original Linked List:")
print_linked_list(head)

head = insert_at_position(head, 2, 10)
print("\nAfter Inserting at Position 2:")
print_linked_list(head)

head = insert_at_end(head, 6)
print("\nAfter Inserting at End:")
print_linked_list(head)

head = delete_at_position(head, 3)
print("\nAfter Deleting at Position 3:")
print_linked_list(head)

head = delete_node_with_value(head, 4)
print("\nAfter Deleting Node with Value 4:")
print_linked_list(head)

head = reverse_linked_list(head)
print("\nReversed Linked List:")
print_linked_list(head)
```
